p2p.gossip

GossipNode:

  • Initialized using (ip, port)
  • Uses a Kademlia DHT to discover peers. DHT uses port+1000
  • Broadcasts and receives messages from the network over udp (port)
  • Messages are encoded as protobuf GossipMessage. (See /protobufs/)
  • Messages can contain new blocks (blockchain), or transactions (mempool)

broadcast_loop() currently broadcasts a random transaction every 10 seconds.

 31class GossipNode:
 32    def __init__(self, address, bootstrap_addr=None, is_full_node=True):
 33        """
 34        Initialize a GossipNode instance.
 35
 36        `is_full_node` is currently used to start a dummy simullation, where
 37        `broadcast_loop()` sends random transactions to the network.
 38
 39        :param address: tuple (host, port), the address of the node.
 40        :param bootstrap_addr: tuple (host, port), optional. The address of the bootstrap node. Default is None.
 41        :param is_full_node: bool, optional. Whether the node is a full node. Default is True.
 42        """
 43        self._address = address  # (host, port)
 44        self._bootstrap_addr = bootstrap_addr
 45        self._is_full_node = is_full_node
 46        self._sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
 47
 48        # Use our custom KademliaNode wrapper
 49        self._dht_node = KademliaNode(
 50            host=address[0],
 51            port=address[1] + 1000,
 52            bootstrap=bootstrap_addr
 53        )
 54
 55    async def republish_loop(self):
 56        """
 57        Republish the gossip port on DHT, every 60 seconds.
 58        This is a workaround for a kademlia DHT limitation, that affects the bootstrap node.
 59        """
 60        while True:
 61            peer_key = f"peer:{self._address[0]}:{self._address[1]+1000}"
 62            try:
 63                # Register the ip:port for DHT, and the gossip port
 64                data = json.dumps({
 65                    "gossip_port": self._address[1]
 66                })
 67                await self._dht_node.set(peer_key, data)
 68                print(f"[âœ“] Set key={peer_key} value={data}")
 69            except Exception as e:
 70                print(f"DHT set error: {e}")
 71            await asyncio.sleep(60)
 72
 73    async def gossip_block(self, block: Block):
 74        """
 75        Gossip a block to all peers in the DHT.
 76        """
 77        message = GossipMessage(
 78            type = GossipMessageType.BLOCK,
 79            block=block,
 80        )
 81        message_bytes = message.SerializeToString()
 82        peers = self._dht_node.get_peers()
 83        for peer in peers:
 84            node_data = await self._dht_node.get(f"peer:{peer[0]}:{peer[1]}")
 85            if not node_data:
 86                print(f"[!] Failed to get gossip port for {peer}")
 87                continue
 88            try:
 89                node_json = json.loads(node_data)
 90                port = node_json["gossip_port"]
 91                self._sock.sendto(message_bytes, (peer[0], port))
 92                print(f"[>] {self._address} Gossiped block to {peer[0]}:{port}, hash={block.hash.hex()}")
 93            except Exception as e:
 94                print(f"[!] Failed to gossip to {peer}: {e}")
 95
 96    async def gossip_transaction(self, transaction: Transaction, exclude_peers=()):
 97        """
 98        Gossip a transaction to all peers in the network.
 99        """
100        message = GossipMessage(
101            type = GossipMessageType.TRANSACTION,
102            transaction=transaction,
103        )
104        message_bytes = message.SerializeToString()
105        peers = self._dht_node.get_peers()
106        for peer in peers:
107            node_data = await self._dht_node.get(f"peer:{peer[0]}:{peer[1]}")
108            if not node_data:
109                print(f"[!] Failed to get gossip port for {peer}")
110                continue
111            try:
112                node_json = json.loads(node_data)
113                port = node_json["gossip_port"]
114                if f"{peer[0]}:{port}" in exclude_peers:
115                    continue
116                self._sock.sendto(message_bytes, (peer[0], port))
117                tx_hash = calculate_tx_hash(transaction)
118                print(f"[<] Gossiped transaction to {peer[0]}:{port}, hash={tx_hash.hex()}")
119            except Exception as e:
120                print(f"[!] Failed to gossip to {peer}: {e}")
121
122    async def listen_for_messages(self):
123        """
124        Listen for incoming messages on the gossip socket.
125        """
126        loop = asyncio.get_running_loop()
127        while True:
128            data, addr = await loop.sock_recvfrom(self._sock, BUF_SIZE)
129            message = GossipMessage()
130            try:
131                message.ParseFromString(data)
132                if message.type == GossipMessageType.BLOCK:
133                    print(f"[>] {self._address} received block from {addr}: hash={message.block.hash.hex()}")
134                    if db.block_exists(message.block.hash):
135                        print(f"[!] Block {message.block.hash.hex()} already exists")
136                    else:
137                        db.block_set(message.block)
138                if message.type == GossipMessageType.TRANSACTION:
139                    tx_hash = calculate_tx_hash(message.transaction)
140                    print(f"[>] Received transaction from {addr[0]}:{addr[1]}: hash={tx_hash.hex()}")
141                    if mempool.hash_exists(tx_hash):
142                        print(f"[!] Transaction {tx_hash.hex()} already in mempool")
143                    else:
144                        mempool.add(message.transaction)
145                        print(f"[âœ“] Added transaction {tx_hash.hex()} to mempool")
146                        asyncio.create_task(self.gossip_transaction(message.transaction, exclude_peers=(f"{addr[0]}:{addr[1]}",)))
147            except Exception as e:
148                print(f"[!] Failed to parse block from {addr}: {e}")
149
150    async def broadcast_loop(self):
151        """
152        Broadcast transactions and blocks to peers.
153        Currently a simulation with random transactions.
154        """
155        while True:
156            # block = random_block(height=10)
157            transaction = random_transaction()
158            asyncio.create_task(self.gossip_transaction(transaction))
159            await asyncio.sleep(10)
160
161    async def run(self):
162        """
163        Run the Gossip node.
164        """
165        self._sock.bind(self._address)
166        self._sock.setblocking(False)
167        print(f"Listening on {self._address}")
168
169        await self._dht_node.start()
170        asyncio.create_task(self.republish_loop())
171        asyncio.create_task(self.listen_for_messages())
172        if self._is_full_node:
173            asyncio.create_task(self.broadcast_loop())
174        while True:
175            await asyncio.sleep(3600)
